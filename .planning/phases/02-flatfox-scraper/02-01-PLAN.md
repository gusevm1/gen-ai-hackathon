    Then create `src/scrapers/flatfox/adapter.ts`:

    ```typescript
    import type { Logger } from 'pino';
    import type { ScraperAdapter, ScrapeOptions, RawRecord } from '../types.js';

    export class FlatfoxAdapter implements ScraperAdapter {
      readonly name = 'flatfox';
      private readonly baseUrl = 'https://flatfox.ch/api/v1/public-listing/';

      constructor(private readonly logger: Logger) {}

      async scrape(options: ScrapeOptions): Promise<RawRecord[]> {
        const allResults: RawRecord[] = [];
        let url: string | null = `${this.baseUrl}?limit=100&expand=images`;

        if (options.dryRun) {
          url = `${this.baseUrl}?limit=10&expand=images`;
        }

        while (url) {
          const response = await fetch(url);
          if (!response.ok) {
            throw new Error(`FlatFox API error: ${response.status} ${response.statusText}`);
          }
          const data = await response.json();
          allResults.push(...(data.results as RawRecord[]));
          this.logger.info(
            { fetched: allResults.length, total: data.count },
            'FlatFox pagination progress',
          );
          url = options.dryRun ? null : data.next;
        }

        return allResults;
      }
    }
    ```

    Key points for the adapter:
    - Follow `data.next` URL as-is for pagination (the API provides full URLs)
    - Stop when `data.next` is null (last page)
    - In dry-run mode, fetch only first page with `limit=10` and stop
    - Always include `expand=images` to get image URLs (without it, images are just PK integers)
    - Log pagination progress with fetched count and total count
    - No retry logic or delays in v1 -- add if Cloudflare blocks during testing
