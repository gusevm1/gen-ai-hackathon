Backend v2 enrichment_status values:
- "available" -- ListingProfile exists, full hybrid scoring used
- "fallback" -- Claude fallback pipeline used (ALLOW_CLAUDE_FALLBACK=true)
- "unavailable" -- No data, no fallback; overall_score=0, match_tier="poor"
- null -- Legacy v1 response or enrichment_status not set

IMPORTANT: The extension receives the FastAPI JSON response directly (camelCase via Pydantic alias_generator).
However, based on the scoring router code, the ScoreResponse model's v2 fields use snake_case in the model definition
and the FastAPI response serialization uses the alias (camelCase). But the field names overall_score, match_tier,
summary_bullets were already snake_case in v1 and work fine. For v2 additions: schema_version stays schema_version
(no camelCase alias changes a single word), criteria_results becomes criteriaResults, enrichment_status becomes
enrichmentStatus. BUT -- looking at the actual TIER_COLORS and existing extension code, the extension already
uses snake_case field names (overall_score, match_tier, etc.) which means the edge function or API proxy
is either returning snake_case OR the extension handles both. Since the existing code works with snake_case
field names in ScoreResponse, keep using snake_case for the new fields too. The edge function passes through
the backend response which is serialized by FastAPI -- need to verify. SAFEST approach: use snake_case field
names (matching what's in the existing interface) and add camelCase aliases as comments for reference.

describe('v2 ScoreResponse shape', () => {
  it('a v2 ScoreResponse with criteria_results conforms to expected shape', () => {
    const sample: ScoreResponse = {
      overall_score: 65,
      match_tier: 'fair',
      summary_bullets: ['Near transit', 'Above budget', 'Good size'],
      categories: [],      // Empty for v2
      checklist: [],        // Empty for v2
      language: 'en',
      schema_version: 2,
      criteria_results: [
        { criterion_name: 'Budget', fulfillment: 0.8, importance: 'critical', weight: 5, reasoning: 'Under budget by 5%' },
        { criterion_name: 'Balcony', fulfillment: 1.0, importance: 'medium', weight: 2, reasoning: 'Has south-facing balcony' },
        { criterion_name: 'Quiet area', fulfillment: null, importance: 'low', weight: 1, reasoning: null },
      ],
      enrichment_status: 'available',
    };
    // Verify types compile and values match
    expect(sample.schema_version).toBe(2);
    expect(sample.criteria_results).toHaveLength(3);
    expect(sample.criteria_results![0].criterion_name).toBe('Budget');
    expect(sample.criteria_results![2].fulfillment).toBeNull();
    expect(sample.enrichment_status).toBe('available');
  });

  it('v1 ScoreResponse still valid without v2 fields', () => {
    const v1: ScoreResponse = {
      overall_score: 82,
      match_tier: 'good',
      summary_bullets: ['Good', 'Fair', 'OK'],
      categories: [{ name: 'location', score: 90, weight: 30, reasoning: ['Near transit'] }],
      checklist: [{ criterion: 'Balcony', met: true, note: 'Yes' }],
      language: 'en',
    };
    expect(v1.schema_version).toBeUndefined();
    expect(v1.criteria_results).toBeUndefined();
    expect(v1.enrichment_status).toBeUndefined();
  });

  it('enrichment_status=unavailable produces valid ScoreResponse', () => {
    const unavailable: ScoreResponse = {
      overall_score: 0,
      match_tier: 'poor',
      summary_bullets: ['Scoring not available', 'Area not yet enriched', 'Check back later'],
      categories: [],
      checklist: [],
      language: 'en',
      schema_version: 2,
      criteria_results: [],
      enrichment_status: 'unavailable',
    };
    expect(unavailable.enrichment_status).toBe('unavailable');
    expect(unavailable.overall_score).toBe(0);
  });
});
