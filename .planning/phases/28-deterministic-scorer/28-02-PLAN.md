    1. score_proximity_quality(
           field: DynamicField,

    2. _score_size_symmetric(
           actual: Optional[float],
           target_min: float,
           target_max: Optional[float],

    3. synthesize_builtin_results(
           prefs: UserPreferences,
           listing: FlatfoxListing,
       ) -> list[FulfillmentResult]:  [DS-06]

       results: list[FulfillmentResult] = []

       # Budget
       if prefs.budget_max is not None:

               actual_rooms = None

       return results
